#include<stdio.h>
#include<conio.h>
#include<graphics.h>
#include<math.h>
int main()
{
    int x1,y1,x2,y2,tx,ty,x3,y3,x4,y4;
    printf("ASHMITA BASNYAT");
    printf("Enter the starting point of the line segment");
    scanf("%d%d",&x1,&y1);
    printf("Enter the ending points of line segments");
    scanf("%d%d",&x2,&y2);
    printf("Enter translation distances tx,ty:\n");
    scanf("%d%d",&tx,&ty);
    int gd=DETECT,gm;
    initgraph(&gd,&gm,"");
    setcolor(WHITE);
    line(x1,y1,x2,y2);
    outtextxy(x2+2,y2+2,"Oringinal line");
    x3=x1+tx;
    y3=y1+ty;
    x4=x2+tx;
    y4=y2+ty;
    setcolor(WHITE);
    line(x3,y3,x4,y4);
    outtextxy(x4+2,y4+2,"Line after translation");
    getch();
    closegraph();
    return 0;
}
